Remove StopWords¶

In [4]:
filtered_words = []
with open('Path-to-/stopwords-fr.json',encoding="utf-8") as stop:
    mystopwords = json.load(stop)
stopeng=stopwords.words("english")
stopfr=stopwords.words("french")
for word in words:
    if not word in mystopwords and not word in stopeng and not word in stopfr and word.isalpha():
        filtered_words.append(word)
print("StopWords Removed")

StopWords Removed
